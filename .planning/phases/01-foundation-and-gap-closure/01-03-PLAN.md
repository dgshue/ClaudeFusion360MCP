    ```python
    def get_body_info(design, rootComp, params):
        body = get_body(rootComp, params.get('body_index'))

        edges_info = []
        for i in range(body.edges.count):
            edge = body.edges.item(i)
            label = label_edge(edge, i, body)  # From helpers/selection.py
            edges_info.append({
                "index": i,
                "label": label,
                "length_cm": round(edge.length, 4),
                "type": "linear" if edge.geometry.objectType == "adsk::core::Line3D" else "curved"
            })

        faces_info = []
        for i in range(body.faces.count):
            face = body.faces.item(i)
            label = label_face(face, i, body)  # From helpers/selection.py
            face_type = "planar"
            geo = face.geometry
            if hasattr(geo, 'objectType'):
                type_map = {
                    'adsk::core::Plane': 'planar',
                    'adsk::core::Cylinder': 'cylindrical',
                    'adsk::core::Sphere': 'spherical',
                    'adsk::core::Cone': 'conical',
                    'adsk::core::Torus': 'toroidal',
                }
                face_type = type_map.get(geo.objectType, 'other')
            faces_info.append({
                "index": i,
                "label": label,
                "area_cm2": round(face.area, 4),
                "type": face_type
            })

        bbox = body.boundingBox
        return {
            "success": True,
            "body_name": body.name,
            "edge_count": body.edges.count,
            "face_count": body.faces.count,
            "edges": edges_info,
            "faces": faces_info,
            "bounding_box": {
                "min": [round(bbox.minPoint.x, 4), round(bbox.minPoint.y, 4), round(bbox.minPoint.z, 4)],
                "max": [round(bbox.maxPoint.x, 4), round(bbox.maxPoint.y, 4), round(bbox.maxPoint.z, 4)]
            }
        }
    ```
    Example output format per user decision: `"Face 0 (top, planar, 25cm2)"`, `"Edge 3 (top-front, linear, 5cm)"` -- the label field contains the semantic description, and structured data is also provided.

    **measure (QUERY-02):**
    ```python
    def measure(design, rootComp, params):
        measure_type = params.get('type', 'body')
        body = get_body(rootComp, params.get('body_index'))

        if measure_type == 'body':
            props = body.physicalProperties
            bbox = body.boundingBox
            return {
                "success": True,
                "volume_cm3": round(props.volume, 6),
                "surface_area_cm2": round(props.area, 4),
                "bounding_box": {
                    "min": [round(bbox.minPoint.x, 4), round(bbox.minPoint.y, 4), round(bbox.minPoint.z, 4)],
                    "max": [round(bbox.maxPoint.x, 4), round(bbox.maxPoint.y, 4), round(bbox.maxPoint.z, 4)],
                    "size": [
                        round(bbox.maxPoint.x - bbox.minPoint.x, 4),
                        round(bbox.maxPoint.y - bbox.minPoint.y, 4),
                        round(bbox.maxPoint.z - bbox.minPoint.z, 4)
                    ]
                }
            }
        elif measure_type == 'edge':
            edge_index = params.get('edge_index', 0)
            edge = body.edges.item(edge_index)
            return {"success": True, "length_cm": round(edge.length, 4)}
        elif measure_type == 'face':
            face_index = params.get('face_index', 0)
            face = body.faces.item(face_index)
            return {"success": True, "area_cm2": round(face.area, 4)}
        else:
            raise ValueError(f"Unknown measure type: {measure_type}. Use 'body', 'edge', or 'face'.")
    ```
